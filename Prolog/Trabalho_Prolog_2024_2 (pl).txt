% Regras
regra1(Usuario, Filme) :- filme(Filme,Genero)
,gosta(Usuario,Genero).

regra2(Filme) :- filme(Filme,Genero)
,gosta(Usuario1,Genero)
,gosta(Usuario2,Genero)
,Usuario1 \= Usuario2
,Usuario1 @< Usuario2.

regra3(Filme, Usuario1, Usuario2) :- filme(Filme,Genero)
,gosta(Usuario1,Genero)
,gosta(Usuario2,Genero)
,Usuario1 \= Usuario2
,Usuario1 @< Usuario2.
